# 1. Ancestors
```js
let foo = {name: 'foo'};
let bar = Object.create(foo);
bar.name = 'bar';
let baz = Object.create(bar);
baz.name = 'baz';
let qux = Object.create(baz);
qux.name = 'qux';

foo.ancestors = function() {
  let result = [];
  let object = this;

  while (Object.getPrototypeOf(object) !== Object.prototype) {
    result.push(Object.getPrototypeOf(object).name);
    object = Object.getPrototypeOf(object);
  }

  if (Object.getPrototypeOf(object) === Object.prototype) {
    result.push('Object.prototype')
  } else {
    result.push('null');
  }

  return result;
}

qux.ancestors();  // returns ['baz', 'bar', 'foo', 'Object.prototype']
baz.ancestors();  // returns ['bar', 'foo', 'Object.prototype']
bar.ancestors();  // returns ['foo', 'Object.prototype']
foo.ancestors();  // returns ['Object.prototype']

console.log(Object.getPrototypeOf(foo) === Object.prototype)
```

# 3. Circular Queue
```js
class CircularQueue {
  constructor (bufferSize) {
    this.bufferSize = bufferSize;
    this.queue = new Array(bufferSize).fill(null);
    this.addIndex = 0;
    this.indexHistory = [];
  }

  enqueue(value) {
    this.queue[this.addIndex] = value;

    this.indexHistory.push(this.addIndex);
    if (this.indexHistory.length > this.bufferSize) this.indexHistory.shift();
    this.addIndex += 1;

    if (this.addIndex === this.bufferSize) this.addIndex = 0;
  }

  dequeue() {
    if (this.indexHistory.length === 0) return null;

    let oldestIndex = this.indexHistory[0];
    let oldestValue = this.queue[oldestIndex];

    this.queue[oldestIndex] = null;
    this.indexHistory.shift();

    return oldestValue;
  }
}

let queue = new CircularQueue(3);
console.log(queue.dequeue() === null);

queue.enqueue(1);
queue.enqueue(2);
console.log(queue.dequeue() === 1);

queue.enqueue(3);
queue.enqueue(4);
console.log(queue.dequeue() === 2);

queue.enqueue(5);
queue.enqueue(6);
queue.enqueue(7);
console.log(queue.dequeue() === 5);
console.log(queue.dequeue() === 6);
console.log(queue.dequeue() === 7);
console.log(queue.dequeue() === null);
```
